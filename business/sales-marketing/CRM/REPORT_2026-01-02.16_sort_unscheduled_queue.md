## Code Highlights

### Sort Function
```javascript
sortUnscheduled(prospects, sortBy) {
    return prospects.sort((a, b) => {
        switch (sortBy) {
            case 'tier':
                return (a.tier || 999) - (b.tier || 999);
            case 'icp':
                return (b.icp_score || 0) - (a.icp_score || 0);
            case 'last_contact':
                if (!a.last_call_date) return -1;
                if (!b.last_call_date) return 1;
                return new Date(a.last_call_date) - new Date(b.last_call_date);
            case 'name':
                return (a.org_name || '').localeCompare(b.org_name || '');
            default:
                return 0;
        }
    });
}
```

### Event Listener
```javascript
document.getElementById('unscheduled-sort').addEventListener('change', (e) => {
    this.unscheduledSort = e.target.value;
    localStorage.setItem('crm_unscheduledSort', this.unscheduledSort);
    this.renderUnscheduled();
});
```
