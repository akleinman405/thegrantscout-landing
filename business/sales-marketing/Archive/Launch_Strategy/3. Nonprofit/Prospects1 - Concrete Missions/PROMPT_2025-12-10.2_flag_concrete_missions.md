## Phase 2: Flag Prospects with Concrete Missions

**Goal:** Apply the concrete mission filter from Phase 1 to flag prospects in the database.

**Prerequisite:** Phase 1 complete. Validated filter: beneficiary keywords in org name = 90% clear missions.

---

## Step 1: Add Column

```sql
ALTER TABLE f990_2025.prospects 
ADD COLUMN IF NOT EXISTS has_concrete_mission SMALLINT DEFAULT 0;
```

---

## Step 2: Apply Validated Keyword Filter from Phase 1

```sql
-- Beneficiary keyword filter (90% validated clear missions)
UPDATE f990_2025.prospects
SET has_concrete_mission = 1
WHERE (
    LOWER(org_name) LIKE '%children%'
    OR LOWER(org_name) LIKE '%youth%'
    OR LOWER(org_name) LIKE '%student%'
    OR LOWER(org_name) LIKE '%shelter%'
    OR LOWER(org_name) LIKE '%homeless%'
    OR LOWER(org_name) LIKE '%habitat%'
    OR LOWER(org_name) LIKE '%food bank%'
    OR LOWER(org_name) LIKE '%meals%'
    OR LOWER(org_name) LIKE '%rescue%'
    OR LOWER(org_name) LIKE '%animal%'
    OR LOWER(org_name) LIKE '%humane%'
    OR LOWER(org_name) LIKE '%fire%'
    OR LOWER(org_name) LIKE '%museum%'
    OR LOWER(org_name) LIKE '%veterans%'
    OR LOWER(org_name) LIKE '%senior%'
    OR LOWER(org_name) LIKE '%disabled%'
    OR LOWER(org_name) LIKE '%hospice%'
)
-- Exclude supporting institutions (tend to be vague)
AND LOWER(org_name) NOT LIKE '%foundation'
AND LOWER(org_name) NOT LIKE '%supporting organization%';
```

## Step 3: Output Summary Stats

```sql
-- Total flagged
SELECT 
    has_concrete_mission,
    COUNT(*) as count,
    ROUND(100.0 * COUNT(*) / SUM(COUNT(*)) OVER (), 1) as pct
FROM f990_2025.prospects
GROUP BY has_concrete_mission;

-- By sector
SELECT 
    sector,
    SUM(has_concrete_mission) as concrete_count,
    COUNT(*) as total,
    ROUND(100.0 * SUM(has_concrete_mission) / COUNT(*), 1) as concrete_pct
FROM f990_2025.prospects
GROUP BY sector
ORDER BY concrete_pct DESC;

-- By employee band (KEY ANALYSIS: reachability vs clarity tradeoff)
-- Phase 1 found: 16-50 employees = 91% clear, but 0-3 = more reachable
SELECT 
    CASE 
        WHEN employee_count = 0 THEN '0 (volunteer only)'
        WHEN employee_count <= 3 THEN '1-3 (most reachable)'
        WHEN employee_count <= 10 THEN '4-10'
        WHEN employee_count <= 15 THEN '11-15'
        WHEN employee_count <= 50 THEN '16-50 (highest clarity)'
        ELSE '51+'
    END as emp_band,
    SUM(has_concrete_mission) as concrete_count,
    COUNT(*) as total,
    ROUND(100.0 * SUM(has_concrete_mission) / COUNT(*), 1) as concrete_pct
FROM f990_2025.prospects
GROUP BY 1
ORDER BY 1;

-- Direct comparison: reachable (0-3) vs high-clarity (16-50)
SELECT 
    CASE 
        WHEN employee_count BETWEEN 0 AND 3 THEN 'Reachable (0-3 emp)'
        WHEN employee_count BETWEEN 16 AND 50 THEN 'High-clarity (16-50 emp)'
        ELSE 'Other'
    END as segment,
    COUNT(*) as total,
    SUM(has_concrete_mission) as with_concrete_mission,
    ROUND(100.0 * SUM(has_concrete_mission) / COUNT(*), 1) as concrete_pct,
    SUM(reach_all_flags) as fully_reachable,
    SUM(received_capacity_grant) as with_capacity_grant
FROM f990_2025.prospects
GROUP BY 1
ORDER BY 1;

-- By revenue band
SELECT 
    revenue_band,
    SUM(has_concrete_mission) as concrete_count,
    COUNT(*) as total,
    ROUND(100.0 * SUM(has_concrete_mission) / COUNT(*), 1) as concrete_pct
FROM f990_2025.prospects
GROUP BY revenue_band
ORDER BY revenue_band;

-- By ICP score
SELECT 
    icp_score,
    SUM(has_concrete_mission) as concrete_count,
    COUNT(*) as total
FROM f990_2025.prospects
GROUP BY icp_score
ORDER BY icp_score DESC;

-- By priority tier
SELECT 
    priority_tier,
    SUM(has_concrete_mission) as concrete_count,
    COUNT(*) as total,
    ROUND(100.0 * SUM(has_concrete_mission) / COUNT(*), 1) as concrete_pct
FROM f990_2025.prospects
GROUP BY priority_tier
ORDER BY priority_tier;
```

## Step 4: Cross-Reference with Other Flags

```sql
-- Concrete + capacity grant
SELECT 
    has_concrete_mission,
    received_capacity_grant,
    COUNT(*) as count
FROM f990_2025.prospects
GROUP BY has_concrete_mission, received_capacity_grant
ORDER BY 1, 2;

-- Concrete + all reachability flags
SELECT 
    has_concrete_mission,
    reach_all_flags,
    COUNT(*) as count
FROM f990_2025.prospects
GROUP BY has_concrete_mission, reach_all_flags
ORDER BY 1, 2;

-- The golden overlap: concrete + reachable + capacity grant
SELECT COUNT(*) as prime_prospects
FROM f990_2025.prospects
WHERE has_concrete_mission = 1
  AND reach_all_flags = 1
  AND received_capacity_grant = 1;
```
