### Query 1: Nonprofits Similar to Beta Clients

```sql
-- Find nonprofits matching beta client profile
-- Example: Senior services in California (like SNS)

WITH beta_profile AS (
    SELECT
        'P' as ntee_code_prefix,  -- Human Services
        'CA' as target_state,
        500000 as min_revenue,
        5000000 as max_revenue
)
SELECT
    nr.organization_name,
    nr.city,
    nr.state,
    nr.total_revenue,
    nr.total_employees_cnt,
    nr.website,
    nr.phone
FROM f990_2025.nonprofit_returns nr
CROSS JOIN beta_profile bp
WHERE nr.state = bp.target_state
  AND nr.total_revenue BETWEEN bp.min_revenue AND bp.max_revenue
  AND nr.ntee_code LIKE bp.ntee_code_prefix || '%'
  AND nr.tax_year >= 2022
ORDER BY nr.total_revenue DESC
LIMIT 500;
```

**Expected yield:** 500-1,000 per beta client profile

### Query 2: Grant-Ready but Under-Resourced

```sql
-- Mid-market nonprofits with revenue but few/no foundation grants

WITH nonprofit_profile AS (
    SELECT
        ein,
        organization_name,
        state,
        city,
        MAX(total_revenue) as revenue,
        MAX(website) as website,
        MAX(phone) as phone,
        MAX(mission_description) as mission
    FROM f990_2025.nonprofit_returns
    WHERE total_revenue BETWEEN 500000 AND 5000000
      AND tax_year >= 2022
    GROUP BY ein, organization_name, state, city
),
grants_received AS (
    SELECT
        recipient_ein,
        COUNT(*) as grant_count
    FROM f990_2025.fact_grants
    WHERE recipient_ein IS NOT NULL
    GROUP BY recipient_ein
)
SELECT
    np.organization_name,
    np.city,
    np.state,
    np.revenue,
    np.website,
    np.phone,
    COALESCE(gr.grant_count, 0) as prior_grants,
    np.mission
FROM nonprofit_profile np
LEFT JOIN grants_received gr ON np.ein = gr.recipient_ein
WHERE COALESCE(gr.grant_count, 0) <= 3
ORDER BY np.revenue DESC
LIMIT 1000;
```

**Total pool:** 108,434 nonprofits matching criteria

### Query 3: Capacity Building Grant Recipients

```sql
-- Nonprofits that received capacity-building grants
-- These have already invested in infrastructure

SELECT DISTINCT
    fg.recipient_name_raw as organization,
    fg.recipient_state as state,
    fg.recipient_city as city,
    COUNT(*) as capacity_grants_received,
    SUM(fg.amount) as total_capacity_funding,
    MAX(fg.tax_year) as most_recent_year
FROM f990_2025.fact_grants fg
WHERE (LOWER(fg.purpose_text) LIKE '%capacity%'
   OR LOWER(fg.purpose_text) LIKE '%organizational development%'
   OR LOWER(fg.purpose_text) LIKE '%technical assistance%'
   OR LOWER(fg.purpose_text) LIKE '%strategic planning%')
GROUP BY fg.recipient_name_raw, fg.recipient_state, fg.recipient_city
HAVING SUM(fg.amount) BETWEEN 10000 AND 500000
ORDER BY total_capacity_funding DESC
LIMIT 500;
```

**Total pool:** ~5,000 unique organizations with capacity-building grants

### Query 4: Grantees of Beta Clients' Known Funders

```sql
-- Find other grantees of foundations that fund your beta clients
-- Replace '123456789' with actual foundation EINs

WITH beta_client_funders AS (
    SELECT DISTINCT foundation_ein
    FROM f990_2025.fact_grants
    WHERE recipient_ein IN ('BETA_CLIENT_EIN_1', 'BETA_CLIENT_EIN_2')
       OR recipient_name_raw ILIKE '%Senior Network Services%'
)
SELECT
    fg.recipient_name_raw as organization,
    fg.recipient_state,
    df.name as funder_name,
    fg.amount,
    fg.purpose_text,
    fg.tax_year
FROM f990_2025.fact_grants fg
JOIN beta_client_funders bcf ON fg.foundation_ein = bcf.foundation_ein
JOIN f990_2025.dim_foundations df ON fg.foundation_ein = df.ein
WHERE fg.recipient_name_raw NOT ILIKE '%Senior Network Services%'  -- Exclude beta client
ORDER BY fg.amount DESC
LIMIT 500;
```
