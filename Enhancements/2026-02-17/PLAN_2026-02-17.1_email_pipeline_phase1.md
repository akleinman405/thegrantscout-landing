### 1c. Populate — Patient assistance foundations (automated)

```sql
INSERT INTO f990_2025.ref_foundation_email_exclusions
    (foundation_ein, foundation_name, exclusion_reason, exclusion_category, notes)
SELECT DISTINCT
    cfp.ein,
    df.name,
    'Patient assistance foundation - gives to individuals not organizations',
    'patient_assistance',
    'total_giving_5yr: ' || cfp.total_giving_5yr::text
FROM f990_2025.calc_foundation_profiles cfp
JOIN f990_2025.dim_foundations df ON df.ein = cfp.ein
WHERE (df.name ILIKE '%patient assistance%'
    OR df.name ILIKE '%patient access%'
    OR df.name ILIKE '%patient foundation%')
  AND cfp.unique_recipients_5yr <= 30
ON CONFLICT DO NOTHING;
```

### 1e. Populate — Geographic monopolies (automated)

```sql
INSERT INTO f990_2025.ref_foundation_email_exclusions
    (foundation_ein, foundation_name, exclusion_reason, exclusion_category, notes)
SELECT
    fsd_top.foundation_ein,
    df.name,
    'Geographic monopoly: ' || ROUND(fsd_top.max_pct * 100) || '% giving in ' || fsd_top.top_state,
    'geographic_monopoly',
    'Should only appear in ' || fsd_top.top_state || ' cohorts. Assets: $' || fp.total_assets_eoy_amt::text
FROM (
    SELECT
        foundation_ein,
        recipient_state AS top_state,
        state_pct AS max_pct,
        ROW_NUMBER() OVER (PARTITION BY foundation_ein ORDER BY state_pct DESC) AS rn
    FROM f990_2025.stg_foundation_state_dist
    WHERE recipient_state != '' AND recipient_state IS NOT NULL
) fsd_top
JOIN f990_2025.dim_foundations df ON df.ein = fsd_top.foundation_ein
JOIN f990_2025.foundation_prospects2 fp ON fp.ein = fsd_top.foundation_ein
WHERE fsd_top.rn = 1
  AND fsd_top.max_pct >= 0.60
  AND fp.total_assets_eoy_amt >= 1000000000
  AND fsd_top.foundation_ein NOT IN (SELECT foundation_ein FROM f990_2025.ref_foundation_email_exclusions)
ON CONFLICT DO NOTHING;
```

## Step 2: Create Geographic Relevance View (15 min)

```sql
CREATE MATERIALIZED VIEW IF NOT EXISTS f990_2025.mv_foundation_geo_relevance AS
SELECT
    fsd.foundation_ein,
    fsd.recipient_state AS state,
    fsd.state_pct,
    fsd.grant_count,
    fsd.total_grants AS state_giving_dollars,
    CASE
        WHEN fsd.state_pct >= 0.25 THEN 'primary'
        WHEN fsd.state_pct >= 0.05 THEN 'secondary'
        WHEN fsd.state_pct >= 0.01 AND fsd.grant_count >= 5 THEN 'tertiary'
        ELSE 'incidental'
    END AS geo_tier,
    CASE
        WHEN max_pct.max_state_pct < 0.15 THEN true
        ELSE false
    END AS is_national_funder
FROM f990_2025.stg_foundation_state_dist fsd
JOIN (
    SELECT foundation_ein, MAX(state_pct) AS max_state_pct
    FROM f990_2025.stg_foundation_state_dist
    WHERE recipient_state != '' AND recipient_state IS NOT NULL
    GROUP BY foundation_ein
) max_pct ON max_pct.foundation_ein = fsd.foundation_ein
WHERE fsd.recipient_state != '' AND fsd.recipient_state IS NOT NULL;

CREATE UNIQUE INDEX ON f990_2025.mv_foundation_geo_relevance (foundation_ein, state);
```

## Step 3: Add Prospect Quality Tiers to np2 (20 min)

```sql
ALTER TABLE f990_2025.nonprofits_prospects2
    ADD COLUMN IF NOT EXISTS email_quality_tier VARCHAR(1),
    ADD COLUMN IF NOT EXISTS email_quality_flags TEXT[];

UPDATE f990_2025.nonprofits_prospects2 np SET
    email_quality_flags = ARRAY_REMOVE(ARRAY[
        CASE WHEN np.contact_first_name IN ('there','Hi','Hello','Dear') THEN 'placeholder_name' END,
        CASE WHEN np.contact_email LIKE '%.gov' OR np.contact_email LIKE '%.gov.%' THEN 'gov_email' END,
        CASE WHEN np.most_recent_tax_year < 2022 THEN 'stale_filer' END,
        CASE WHEN np.ntee_code IS NULL THEN 'missing_ntee' END,
        CASE WHEN np.contact_email LIKE 'info@%' OR np.contact_email LIKE 'contact@%'
            OR np.contact_email LIKE 'office@%' OR np.contact_email LIKE 'admin@%'
            OR np.contact_email LIKE 'team@%' THEN 'generic_email' END
    ], NULL),
    email_quality_tier = CASE
        WHEN np.contact_first_name NOT IN ('there','Hi','Hello','Dear')
            AND np.contact_email NOT LIKE 'info@%'
            AND np.contact_email NOT LIKE 'contact@%'
            AND np.contact_email NOT LIKE 'office@%'
            AND np.contact_email NOT LIKE 'admin@%'
            AND np.contact_email NOT LIKE 'team@%'
            AND np.contact_email NOT LIKE '%.gov'
            AND np.most_recent_tax_year >= 2022
            AND np.ntee_code IS NOT NULL
        THEN 'A'
        WHEN np.most_recent_tax_year >= 2022
            AND np.ntee_code IS NOT NULL
            AND np.contact_email NOT LIKE '%.gov'
            AND (np.contact_first_name NOT IN ('there','Hi','Hello','Dear')
                 OR (np.contact_email NOT LIKE 'info@%'
                     AND np.contact_email NOT LIKE 'contact@%'
                     AND np.contact_email NOT LIKE 'office@%'))
        THEN 'B'
        WHEN np.most_recent_tax_year >= 2022
            AND np.contact_email NOT LIKE '%.gov'
        THEN 'C'
        ELSE 'D'
    END
WHERE np.contact_email IS NOT NULL;
```

FROM f990_2025.nonprofits_prospects2

## Step 4: Create Pre-Flight Quality Views (30 min)

### 4a. Prospect quality view

```sql
CREATE OR REPLACE VIEW f990_2025.v_email_preflight_prospect_quality AS
SELECT
    np.ein, np.organization_name, np.state, np.ntee_code,
    np.contact_email, np.contact_first_name,
    np.email_quality_tier, np.email_quality_flags,
    CASE WHEN np.contact_first_name IN ('there','Hi','Hello','Dear','Sir','Madam','whom','Team')
         THEN true ELSE false END AS flag_placeholder_name,
    CASE WHEN np.contact_email LIKE 'info@%' OR np.contact_email LIKE 'contact@%'
         OR np.contact_email LIKE 'office@%' OR np.contact_email LIKE 'admin@%'
         OR np.contact_email LIKE 'team@%'
         THEN true ELSE false END AS flag_generic_email,
    CASE WHEN np.contact_email LIKE '%.gov' OR np.contact_email LIKE '%.gov.%'
         THEN true ELSE false END AS flag_gov_email,
    CASE WHEN np.most_recent_tax_year < 2022
         THEN true ELSE false END AS flag_stale_filer,
    CASE WHEN np.ntee_code IS NULL
         THEN true ELSE false END AS flag_missing_ntee,
    CASE WHEN np.mission_description IS NULL OR LENGTH(np.mission_description) < 10
         THEN true ELSE false END AS flag_missing_mission
FROM f990_2025.nonprofits_prospects2 np
WHERE np.contact_email IS NOT NULL;
```

### 4b. Foundation over-citation view

```sql
CREATE OR REPLACE VIEW f990_2025.v_email_preflight_overcitation AS
SELECT
    cfl.foundation_ein,
    cfl.foundation_name,
    COUNT(DISTINCT (cfl.state, cfl.ntee_sector)) AS cohort_appearances,
    ARRAY_AGG(DISTINCT cfl.state || '-' || cfl.ntee_sector ORDER BY cfl.state || '-' || cfl.ntee_sector) AS cohorts
FROM f990_2025.cohort_foundation_lists cfl
GROUP BY cfl.foundation_ein, cfl.foundation_name
HAVING COUNT(DISTINCT (cfl.state, cfl.ntee_sector)) > 3;
```

### 4c. Example grant quality view

```sql
CREATE OR REPLACE VIEW f990_2025.v_email_preflight_example_quality AS
SELECT
    cfl.state, cfl.ntee_sector,
    cfl.foundation_name, cfl.example_recipient_name,
    cfl.example_grant_amount,
    dr.ntee_code AS example_ntee,
    CASE
        WHEN dr.ntee_code IS NULL THEN 'unknown_ntee'
        WHEN LEFT(dr.ntee_code, 1) != cfl.ntee_sector THEN 'sector_mismatch'
        ELSE 'ok'
    END AS ntee_match_status
FROM f990_2025.cohort_foundation_lists cfl
LEFT JOIN f990_2025.dim_recipients dr
    ON LOWER(dr.name) = LOWER(cfl.example_recipient_name)
    AND dr.state = cfl.state
WHERE cfl.example_recipient_name IS NOT NULL;
```
